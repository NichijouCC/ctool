    example
    retryFn(() => new Promise((resolve, reject) => {
    const value = Math.random();
    if (value > 0.9) {
        resolve(value);
    } else {
        reject(value);
    }
}), {
    onceTryBefore: (index) => console.log("开始执行", index),
    onceTryCallBack: (index, result, err) => console.log("单次执行结果:", index, result, err)
})
    .then(result => {
        // do somethin
    })
    .catch(err => {
        // do somethin
    });
