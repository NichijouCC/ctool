Class Timer

计时器

example
const timer = Timer.create();
// update监听
timer.tick.addEventListener((deltaTime) => {
    console.log("deltaTime:", deltaTime);
    // 总计时
    const time = timer.recordTime;
});

// 重置计时
timer.resetRecordTime();

// 释放
timer.dispose();
