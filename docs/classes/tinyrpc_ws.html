Class TinyRpc_ws

基于ws的简易rpc实现

example
let ins: TinyRpc_ws = TinyRpc_ws.connect("ws://localhost:3030") as any;
ins.on("connect", () => {
     //调用远程方法
    ins.callMethod("add", [1, 2])
        .then(result => {
            console.log("result equal 3", result);
        })
})
description


消息体为buffer格式;

发送消息结构:

{
     id: "消息id",
     method: "消息名",
     params: "消息参数"
}

返回消息结构:

{
     id:"消息id",
     result:"方法结果"
}
